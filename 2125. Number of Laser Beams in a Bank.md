2125. Number of Laser Beams in a Bank

```js
/**
 * @param {string[]} bank
 * @return {number}
 */
var numberOfBeams = function(bank) {
    const m = bank.length, n = bank[0].length;
    const devices = new Array(m).fill(0);
    let result = 0;
    
    for(let i = 0; i < m; i ++){
        for(let j = 0; j < n; j ++){
            if(bank[i][j] === "1")  devices[i] += 1;
        }
    }

    let prev = devices[0];

    for(let i = 1; i < m; i ++){
        if(devices[i] === 0)    continue;
        if(prev === 0){
            prev = devices[i];
            continue;
        }
        result += prev * devices[i];
        prev = devices[i];
    }   

    return result;
};
```
