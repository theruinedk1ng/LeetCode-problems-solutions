1513. Number of Substrings With Only 1s

# Python Version
```
class Solution:
    def numSub(self, s: str) -> int:

        return sum(len(sub) * (len(sub) + 1) // 2 for sub in s.split('0')) % (10 ** 9 + 7)
```

# JS Version 1
```js
/**
 * @param {string} s
 * @return {number}
 */
var numSub = function(s) {
    
    let auxString = "", result = 0, hasZero = false, n = s.length;

    for(let i = 0; i < n; i ++){
        if(s[i] === "0"){
            hasZero = true;
            result += auxString.length * (auxString.length + 1) / 2;
            auxString = "";
        }
        else{
            auxString += s[i];
        }
    }

    if(hasZero === false)
        return (n*(n + 1) / 2) % ((10**9)+7);

    return (result += auxString.length * (auxString.length + 1) / 2) % ((10**9)+7);
};
```

/**
 * @param {string} s
 * @return {number}
 */
var numSub = function(s) {
    let total = 0;
    s.split("0").filter(block => {
        const length = block.length;
        if(length) {
            total += (length * (length + 1))/2
        }
    });
    return total % (10**9 + 7);
};
