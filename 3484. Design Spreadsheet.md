```js

 * @param {number} rows
 */
var Spreadsheet = function(rows) {
    this.sheet = {};
};

/** 
 * @param {string} cell 
 * @param {number} value
 * @return {void}
 */
Spreadsheet.prototype.setCell = function(cell, value) {
    this.sheet[cell] = value;
};

/** 
 * @param {string} cell
 * @return {void}
 */
Spreadsheet.prototype.resetCell = function(cell) {
    this.sheet[cell] = 0;
};

/** 
 * @param {string} formula
 * @return {number}
 */
Spreadsheet.prototype.getValue = function(formula) {
    let [x, y] = formula.slice(1).split('+');
    let x_val = 0, y_val = 0;

    function isDigit(ch){
        return /^\d+$/.test(ch);
    }

    if(this.sheet.hasOwnProperty(x)){
        x_val = this.sheet[x];
    }
    else{
        if(isDigit(x))
            x_val = parseInt(x);
        else
            x_val = 0;
    }

    if(this.sheet.hasOwnProperty(y)){
        y_val = this.sheet[y];
    }
    else{
        if(isDigit(y))
            y_val = parseInt(y);
        else
            y_val = 0;
    }

    return x_val + y_val;
};
```

```py
class Spreadsheet:

    def __init__(self, rows: int):

        self.spreadsheet = {}
        labels = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'

        for label in labels:
            for i in range(1, rows + 1):
                self.spreadsheet[label + str(i)] = 0
        

    def setCell(self, cell: str, value: int) -> None:

        self.spreadsheet[cell] = value


    def resetCell(self, cell: str) -> None:

        self.spreadsheet[cell] = 0
        

    def getValue(self, formula: str) -> int:
        
        n = len(formula)
        aux = ''
        result = 0

        for i in range(1, n):
            if formula[i] != '+':
                aux += formula[i]
            else:
                if aux.isdigit():
                    result += int(aux)
                else:
                    result += self.spreadsheet[aux]
                aux = ''
                
        if aux.isdigit():
            result += int(aux)
        else:
            result += self.spreadsheet[aux]
        
        return result
```
