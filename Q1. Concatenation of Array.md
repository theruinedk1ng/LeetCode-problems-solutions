Q1. Concatenation of Array

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var getConcatenation = function(nums) {
    
    const n = nums.length;
    let answer = new Array(2*n);

    for (let i = 0; i < 2 * n; i ++) {
        answer[i] = nums[i % n]; 
    }

    return answer;
};
```
```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var getConcatenation = function(nums) {
    return nums.concat(nums);
};
```
```py
class Solution:
    def getConcatenation(self, nums: List[int]) -> List[int]:
        return nums + nums
```
