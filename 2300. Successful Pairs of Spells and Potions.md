2300. Successful Pairs of Spells and Potions

```js
/**
 * @param {number[]} spells
 * @param {number[]} potions
 * @param {number} success
 * @return {number[]}
 */
var successfulPairs = function(spells, potions, success) {
    const n = spells.length, m = potions.length;
    let pairs = new Array(n).fill(0);
    potions.sort((a, b) => a - b);

    function binarySearch(idx){
        let l = 0, r = m - 1;

        while(l <= r){
            const m = Math.floor((l + r) / 2);

            if(spells[idx] * potions[m] < success) l = m + 1;
            else r = m - 1;
        }

        return m - l;
    }

    for(let i = 0; i < n; i ++){
        const count = binarySearch(i);
        pairs[i] = count;
    }

    return pairs;
};
```
