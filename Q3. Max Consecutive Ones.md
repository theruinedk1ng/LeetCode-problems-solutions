Q3. Max Consecutive Ones

```py
class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        
        result = 0
        length = 0

        for num in nums:
            if num == 1:
                length += 1
            else:
                result = max(result, length)
                length = 0
        
        return max(result, length)
```
```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var findMaxConsecutiveOnes = function(nums) {
    const n = nums.length;
    let previous = nums[0], max = -1, len = previous;

    if (!nums.includes(1))
        return 0;

    for (let i = 1; i < n; i ++) {
        if (previous === nums[i] && nums[i] === 1)
            len += 1;
        else {
            max = Math.max(max, len);
            len = 1;
        }
        previous = nums[i];
    };

    return Math.max(max, len);
};
```
