36. Valid Sudoku

```py
class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        
        for i in range(9):
            seenRow = set()
            seenColumn = set()
            for j in range(9):
                if board[i][j] == '.' and board[j][i] == '.':
                    continue
                if board[j][i] != '.':
                    if int(board[j][i]) in seenColumn:
                        return False
                    else:
                        seenColumn.add(int(board[j][i]))
                if board[i][j] != '.':
                    if int(board[i][j]) in seenRow:
                        return False
                    else:
                        seenRow.add(int(board[i][j]))
                        
        for boxRow in range(0, 9, 3):
            for boxCol in range(0, 9, 3):
                seenBox = set()
                for i in range(3):
                    for j in range(3):
                        val = board[boxRow + i][boxCol + j]
                        if val != '.':
                            if val in seenBox:
                                return False
                            seenBox.add(val)
        
        return True
```

```py
class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        
        # check rows & columns
        cells = [[] for _ in range(9)]
        for i in range(9):
            row = []
            column = []
            for j in range(9):
                if board[i][j] != '.':
                    column.append(board[i][j])
                    x = ((i // 3) * 3) + (j // 3)
                    cells[x].append(board[i][j])
                if board[j][i] != '.':
                    row.append(board[j][i])
            if len(set(row)) != len(row) or len(set(column)) != len(column):
                return False
        
        # check cells 
        for cell in cells:
            if len(set(cell)) != len(cell):
                return False


        return True
```
