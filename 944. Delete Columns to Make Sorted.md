944. Delete Columns to Make Sorted

```js
/**
 * @param {string[]} strs
 * @return {number}
 */
var minDeletionSize = function(strs) {
    const n = strs.length, m = strs[0].length;
    let result = 0;

    for(let i = 0; i < m; i ++){
        for(let j = 1; j < n; j ++){
            if(strs[j - 1][i] <= strs[j][i])
                continue;
            result += 1;
            break;
        }
    }

    return result;
};
```
