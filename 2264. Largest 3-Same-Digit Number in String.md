2264. Largest 3-Same-Digit Number in String

```js
/**
 * @param {string} num
 * @return {string}
 */
var largestGoodInteger = function(num) {

    let arr = [], count = 1, n = num.length, prev = num[0], s = prev;


    for(let i = 1; i < n; i ++){
        if(prev === num[i]){
            count += 1;
            s += prev;
        }
        else{
            count = 1;
            s = num[i];
        }
        
        if(count === 3){
            arr.push(parseInt(s, 10));
            s = num[i];
        }

        prev = num[i];
    }

    if(arr.length === 0)
        return '';

    let result = Math.max(...arr);

    if(result === 0)
        return '000';

    return result.toString();
    
};
```

class Solution:
    def largestGoodInteger(self, num: str) -> str:
        
        n = len(num)
        prev = num[0]
        count = 1
        tmp = prev
        numbers = []

        for i in range(1, n):
            if num[i] == prev:
                count += 1
                tmp += prev
            else:
                count = 1
                tmp = num[i]
            
            if count == 3:
                numbers.append(int(tmp))
                tmp = num[i]
            
            prev = num[i]
        
        print(numbers)

        if len(numbers) == 0:
            return ''
        
        result = max(numbers)

        return '000' if result == 0 else str(result)
