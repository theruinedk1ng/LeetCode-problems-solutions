1716. Calculate Money in Leetcode Bank

```js
/**
 * @param {number} n
 * @return {number}
 */
var totalMoney = function(n) {
    let prevMonday = 1, prevDay = 1, result = 0;

    for(let i = 0; i < n; i ++){
        if((i + 1) % 7 === 0) {
            result = result + prevDay;
            prevMonday += 1;
            prevDay = prevMonday;
        }
        else{
            result = result + prevDay;
            prevDay += 1;
        }
    }
    return result;
};
```
