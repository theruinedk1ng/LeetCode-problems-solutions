11. Container With Most Water

```js
/**
 * @param {number[]} height
 * @return {number}
 */
var maxArea = function(height) {
    let n = height.length;
    let i = 0, j = n - 1, result = 0;

    while(i < j){
        result = Math.max(result, Math.min(height[i], height[j]) * (j - i));
        if(height[i] >= height[j])
            j --;
        else if(height[i] < height[j])
            i ++;
    }
    return result;
};
```
