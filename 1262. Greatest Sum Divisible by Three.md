1262. Greatest Sum Divisible by Three

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSumDivThree = function(nums) {

    let dp = [0, -Infinity, -Infinity];

    for (const x of nums) {
        const next = [...dp]; 
        for (let r = 0; r < 3; r++) {
            const newSum = dp[r] + x;
            const newR = newSum % 3;
            next[newR] = Math.max(next[newR], newSum);
        }
        dp = next;
    }

    return dp[0];   
};
```
