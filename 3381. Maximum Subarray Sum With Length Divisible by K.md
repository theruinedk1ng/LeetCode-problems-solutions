3381. Maximum Subarray Sum With Length Divisible by K

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var maxSubarraySum = function(nums, k) {
    const n = nums.length;

    let prefix = 0;

    const minPrefix = new Array(k).fill(Infinity);

    minPrefix[0] = 0;

    let result = -Infinity;

    for (let i = 0; i < n; i++) {
        prefix += nums[i];

        const mod = (i + 1) % k;

        result = Math.max(result, prefix - minPrefix[mod]);

        minPrefix[mod] = Math.min(minPrefix[mod], prefix);
    }

    return result;
};
```
