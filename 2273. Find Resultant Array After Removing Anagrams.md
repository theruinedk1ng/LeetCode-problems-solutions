2273. Find Resultant Array After Removing Anagrams

```js
/**
 * @param {string[]} words
 * @return {string[]}
 */
var removeAnagrams = function(words) {
    let n = words.length;
    let maps = Array.from({ length: n }, () => new Map());

    function mapsAreEqual(map1, map2){
        if(map1.size !== map2.size) return false;
        for(let [key, value] of map1){
            if (!map2.has(key) || map2.get(key) !== value) return false;
        }
        return true;
    }

    for(let i = 0; i < n; i ++){
        for(let j = 0; j < words[i].length; j ++){
            if(maps[i].has(words[i][j]))
                maps[i].set(words[i][j], maps[i].get(words[i][j]) + 1);
            else
                maps[i].set(words[i][j] , 1);
        }
    }   

    for(let i = 1; i < n; i ++){
        if(mapsAreEqual(maps[i], maps[i - 1]))
        {
            words.splice(i, 1);
            maps.splice(i, 1);
            i --;
            n --;
        }
    }

    return words;
};
```
