1437. Check If All 1's Are at Least Length K Places Away

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {boolean}
 */
var kLengthApart = function(nums, k) {
    let prevIndexOfLastOne = -1;

    for(let i = 0; i < nums.length; i ++){
        if(prevIndexOfLastOne === -1){
            if(nums[i] === 1)
                prevIndexOfLastOne = i;
            continue;
        }
        if(nums[i] === 1){
            if(i - prevIndexOfLastOne > k)
                prevIndexOfLastOne = i;
            else   
                return false;
        }
    }

    return true;
};
```
