819. Most Common Word

```js
/**
 * @param {string} paragraph
 * @param {string[]} banned
 * @return {string}
 */
var mostCommonWord = function(paragraph, banned) {
    const re = /[!?',;. ]/;
    let clean = paragraph.split(re), freq = new Map(), max = -1, result = '';
    clean = clean.map(s => s.toLowerCase());

    for (let c of clean) {
        if (c === '')
            continue;
        if (!freq.has(c))
            freq.set(c, 0);
        freq.set(c, freq.get(c) + 1);
        if (!banned.includes(c)){
            max = Math.max(max, freq.get(c));
            if (freq.get(c) === max)
                result = c;
        }
    }

    return result;
};
```

```py
class Solution:
    def mostCommonWord(self, paragraph: str, banned: List[str]) -> str:
         
        delimiters = "!?',;. "
        maximum = -1
        result = ''
        freq = {}
        clean = re.split(f"[{re.escape(delimiters)}]", paragraph)
        clean = [c.lower() for c in clean]

        for c in clean:
            if c == '':
                continue
            if c not in freq:
                freq[c] = 0
            freq[c] += 1
            if c not in banned:
                maximum = max(maximum, freq[c])
                if freq[c] == maximum:
                    result = c  

        return result
```
