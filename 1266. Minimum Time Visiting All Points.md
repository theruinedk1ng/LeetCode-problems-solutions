1266. Minimum Time Visiting All Points

```js
/**
 * @param {number[][]} points
 * @return {number}
 */
var minTimeToVisitAllPoints = function(points) {
    const n = points.length;

    // Initialize the result with 0 and the previous x and y with the points at index 0
    let result = 0, previous_x = points[0][0], previous_y = points[0][1];

    // Iterate through points starting at index 1
    for (let i = 1; i < n; i ++) {
        //  Get the current difference and points
        let x = points[i][0], y = points[i][1];

        // Add to the result the maximum between horizontal distance and vertical distance
        result += Math.max(Math.abs(x - previous_x), Math.abs(y - previous_y));

        // Update previous variables
        previous_x = x;
        previous_y = y;
    };

    return result;
};
```
