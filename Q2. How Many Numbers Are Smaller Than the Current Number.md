Q2. How Many Numbers Are Smaller Than the Current Number

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var smallerNumbersThanCurrent = function(nums) {
    const n = nums.length;
    let result = new Array(n), auxiliary = [...nums], greater = new Map(), freq = new Map();
    auxiliary.sort((a, b) => a - b);

    for (let i = 0; i < n; i ++){
        if (!freq.has(auxiliary[i]))
            freq.set(auxiliary[i], 0);
        freq.set(auxiliary[i], freq.get(auxiliary[i]) + 1);
        greater.set(auxiliary[i], i);
    };

    for (let i = 0; i < n; i ++) {
        if (freq.get(nums[i]) > 1)
            result[i] = greater.get(nums[i]) - freq.get(nums[i]) + 1;
        else
            result[i] = greater.get(nums[i]);
    };

    return result;
};
```
```py
class Solution:
    def smallerNumbersThanCurrent(self, nums: List[int]) -> List[int]:
        
        n = len(nums)
        result = [0] * n

        for i in range(n):
            for j in range(n):
                if nums[j] < nums[i]:
                    result[i] += 1
        
        return result
```
