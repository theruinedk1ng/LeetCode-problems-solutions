2269. Find the K-Beauty of a Number

```js
/**
 * @param {number} num
 * @param {number} k
 * @return {number}
 */
var divisorSubstrings = function(num, k) {

    let digits = num.toString();
    let n = digits.length, i = 0, j = k, result = 0;

    while(j <= n){
        if(parseInt(digits.slice(i, j)) != 0 && num % parseInt(digits.slice(i, j)) === 0)
            result += 1;

        i += 1;
        j += 1;
    }
    
    return result;
};
```

```py
class Solution:
    def divisorSubstrings(self, num: int, k: int) -> int:
        
        i, j = 0, k
        result = 0
        digits = str(num)
        n = len(digits)

        while j <= n:
            if int(digits[i:j]) != 0 and num % int(digits[i:j]) == 0:
                result += 1

            i += 1
            j += 1

        return result
```
