3461. Check If Digits Are Equal in String After Operations I

```js
/**
 * @param {string} s
 * @return {boolean}
 */
var hasSameDigits = function(s) {
    
    while(s.length > 2){
        let aux = '';
        for(let i = 1; i < s.length; i ++){
            aux = aux.concat((parseInt(s[i - 1]) + parseInt(s[i])) % 10).toString();
        }
        s = aux;
    }

    return s[0] === s[1];
};
```
