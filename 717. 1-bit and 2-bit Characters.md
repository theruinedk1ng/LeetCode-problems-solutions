717. 1-bit and 2-bit Characters

```js
/**
 * @param {number[]} bits
 * @return {boolean}
 */
var isOneBitCharacter = function(bits) {
    let prev = bits[0];

    for(let i = 1; i < bits.length; i ++){
        if(prev === 0 || prev.toString().length > 1){
            prev = bits[i];
        }
        else{
            prev = prev * 10 + bits[i];
        }
    }

    return prev === 0 ? true : false;

};
```
