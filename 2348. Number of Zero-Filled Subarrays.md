2348. Number of Zero-Filled Subarrays

```py
class Solution:
    def zeroFilledSubarray(self, nums: List[int]) -> int:
        
        subarrays = 0
        freq = Counter(nums)

        if 0 not in freq:
            return subarrays

        lengthOfSubarray = 1 if nums[0] == 0 else 0
        n = len(nums)
        previous = nums[0]
        
        for i in range(1, n):
            if previous == nums[i] and previous == 0:
                lengthOfSubarray += 1
            else:
                previous = nums[i]
                subarrays += lengthOfSubarray * (lengthOfSubarray + 1) // 2 
                lengthOfSubarray = 1 if previous == 0 else 0

        subarrays += lengthOfSubarray * (lengthOfSubarray + 1) // 2 
            
        return subarrays
```
