1207. Unique Number of Occurrences

```js
/**
 * @param {number[]} arr
 * @return {boolean}
 */
var uniqueOccurrences = function(arr) {
    
    // Time complexity O(n)

    let seen = new Set(), freq = new Map();

    for (const n of arr) {
        if (!freq.has(n))
            freq.set(n, 0);
        freq.set(n, freq.get(n) + 1);
    };

    for (const [key, value] of freq) {
        if (seen.has(value))
            return false;
        seen.add(value);
    };

    return true;
};
```
```py
class Solution:
    def uniqueOccurrences(self, arr: List[int]) -> bool:

        # Time Complexity: O(n)

        freq = Counter(arr)
        seen = set()

        for key, value in freq.items():
            if value in seen:
                return False
            seen.add(value)

        return True
```
