1390. Four Divisors

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var sumFourDivisors = function(nums) {
    const n = nums.length;
    let result = 0, seen = new Map();

    function divisors(x) {
        // we initialize count with 2 taking 1 and the number itself
        // therefore the sum of the divisors will be initalized with the number itself + 1
        let count = 2, sum = x + 1;
        for (let i = 2; i <= Math.sqrt(x); i ++){
            if (x % i === 0) {
                count += 1;
                sum += i;
                // if x / i is different we add it too
                if(i !== x / i){
                    count += 1;
                    sum += x / i;
                }
            }
        };
        return count === 4 ? sum : 0;
    };

    for (let num of nums) {
        result += divisors(num);
    };

    return result;
};
```
