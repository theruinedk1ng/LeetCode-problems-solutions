1957. Delete Characters to Make Fancy String

```js
/**
 * @param {string} s
 * @return {string}
 */
var makeFancyString = function(s) {

    let n = s.length;
    let list = s.split('');
    let freq = 1;
    let prev = list[0];

    for(let i = 1; i < n; i ++){
        if(list[i] == prev){
            freq += 1;
        }
        else{
            freq = 1;
            prev = list[i];
        }

        if(freq == 3){
            freq -= 1;
            list[i - 2] = '';
        }
    }

    return list.join('');

};
```

```py
class Solution:
    def makeFancyString(self, s: str) -> str:
        
        n = len(s)
        s = list(s)
        prev = s[0]
        freq = 1

        for i in range(1, n):
            if s[i] == prev:
                freq += 1
            else:
                freq = 1
                prev = s[i]
            
            if freq == 3:
                freq -= 1
                s[i - 2] = ''
        
        return ''.join(s)
```
