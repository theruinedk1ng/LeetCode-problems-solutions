342. Power of Four

```js
/**
 * @param {number} n
 * @return {boolean}
 */
var isPowerOfFour = function(n) {

    if(n <= 0)
        return false
    
    function four(x){
        if(x === 1)
            return true
        if(x % 4 !== 0)
            return false
        return four(x / 4)
    }

    return four(n);

};
```
