1200. Minimum Absolute Difference

```js
/**
 * @param {number[]} arr
 * @return {number[][]}
 */
var minimumAbsDifference = function(arr) {
    const n = arr.length;
    // Initialize the minimum with infinity and the result array
    let minimum = Infinity, result = [];
    // Sort the array in an increasing order
    // so that the pairs will follow the constraint a < b
    arr.sort((a, b) => a - b);

    // Find the minimum absolute difference
    for (let i = 1; i < n; i ++) {
        minimum = Math.min(minimum, Math.abs(arr[i] - arr[i - 1]));
    };

    // Iterate again through the elements of the array
    for (let i = 1; i < n ; i ++) {
        /* If we find a pair that its absolute difference
        that is equal to the minimum absolute difference of any two elements
        we add it to the result array */
        if (Math.abs(arr[i] - arr[i - 1]) === minimum)
            result.push([arr[i - 1], arr[i]]);
    };

    return result;
};
```
