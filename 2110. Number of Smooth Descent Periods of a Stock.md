2110. Number of Smooth Descent Periods of a Stock

```js
/**
 * @param {number[]} prices
 * @return {number}
 */
var getDescentPeriods = function(prices) {
    let ans = 0;
    let len = 1;

    for (let i = 1; i <= prices.length; i++) {
        if (i < prices.length && prices[i - 1] - prices[i] === 1) {
            len++;
        } else {
            ans += (len * (len + 1)) / 2;
            len = 1;
        }
    }

    return ans;
};
```
