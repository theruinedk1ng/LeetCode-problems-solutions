3370. Smallest Number With All Set Bits

```js
/**
 * @param {number} n
 * @return {number}
 */
var smallestNumber = function(n) {

    function onlySetBits(x){
        for(let b of x){
            if(b === "0")
                return false;
        }
        return true;
    }

    while(onlySetBits(n.toString(2)) === false){
        n += 1;
    }

    return n;
};
```
