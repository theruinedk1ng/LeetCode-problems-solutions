414. Third Maximum Number

```py
class Solution:
    def thirdMax(self, nums: List[int]) -> int:

        nums.sort(reverse = True)
        visited = {num: False for num in nums}
        counter = 0

        for num in nums:
            if not visited[num]:
                counter += 1
                visited[num] = True
            if counter == 3:
                return num
        
        return max(nums)
```

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var thirdMax = function(nums) {

    let counter = 0, visited = {};
    nums.sort((a, b) => b - a);
    
    for(let num of nums){
        visited[num] = false;
    }

    for(let num of nums){
        if(visited[num] === false){
            counter += 1;
            visited[num] = true;
        }
        if(counter === 3)
            return num;
    }

    return Math.max(...nums);
};
```
