2011. Final Value of Variable After Performing Operations

```js
/**
 * @param {string[]} operations
 * @return {number}
 */
var finalValueAfterOperations = function(operations) {
    let x = 0;
    const incrementOperations = ['++X', 'X++'];
    const decrementOperations = ['--X', 'X--'];

    for(let operation of operations)
    {
        if(incrementOperations.includes(operation)) x += 1;
        else if(decrementOperations.includes(operation)) x-= 1;
    }

    return x;
};
```
