Q1. Build an Array With Stack Operations

```js
/**
 * @param {number[]} target
 * @param {number} n
 * @return {string[]}
 */
var buildArray = function(target, n) {
    let operations = [], stack = [];

    for (let i = 1; i <= n; i ++) {
        operations.push('Push');
        stack.push(i);
        if (!target.includes(i)) {
            operations.push('Pop');
            stack.pop();
        }
        if (JSON.stringify(stack) === JSON.stringify(target))
            return operations;
    };

    return operations;
};
```
```py
class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        
        operations = []
        stack = []

        for i in range(1, n + 1):
            operations.append('Push')
            stack.append(i)
            if i not in target:
                operations.append('Pop')
                stack.pop()
            if stack == target:
                return operations
        
        return operations
```
