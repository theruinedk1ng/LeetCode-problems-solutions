3074. Apple Redistribution into Boxes

```js
/**
 * @param {number[]} apple
 * @param {number[]} capacity
 * @return {number}
 */
var minimumBoxes = function(apple, capacity) {
    const n = apple.length, m = capacity.length;
    let result = 0, appleCount = apple.reduce((a, b) => a + b, 0), s = 0;
    capacity.sort((a, b) => b - a);

    for (let i = 0; i < m; i ++) {
        s += capacity[i];
        result += 1;
        if (s >= appleCount) 
            return result;
    };

    return result;
};
```
```py
class Solution:
    def minimumBoxes(self, apple: List[int], capacity: List[int]) -> int:
        
        appleCount = sum(apple)
        capacity.sort(reverse = True)
        result = s = 0

        for c in capacity:
            result += 1
            s += c
            if s >= appleCount:
                return result

        return result
```
