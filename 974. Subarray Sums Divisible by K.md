974. Subarray Sums Divisible by K

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var subarraysDivByK = function(nums, k) {
    const n = nums.length;
    let prefix = 0, rem = new Array(n), result = 0, freq = new Array(k).fill(0);

    for(let i = 0; i < n; i ++){
        prefix += nums[i];
        rem[i] = ((prefix % k) + k) % k;

    }

    freq[0] = 1;

    for(let i = 0; i < n; i ++){
        result += freq[rem[i]];
        freq[rem[i]] += 1;
    }

    return result;
};
```
