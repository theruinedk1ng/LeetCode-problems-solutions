2483. Minimum Penalty for a Shop

```js
/**
 * @param {string} customers
 * @return {number}
 */
var bestClosingTime = function(customers) {
    const n = customers.length;
    let yCount = 0, nCount = 0, result = 0, yCurrent = 0, nCurrent = 0;

    for (let customer of customers) {
        yCount += customer === 'Y' ? 1 : 0;
        nCount += customer === 'N' ? 1 : 0;
    };

    let minPenalty = yCount;

    for (let i = 0; i < n; i ++) {
        if (customers[i] === 'Y')
            yCurrent += 1;
        else
            nCurrent += 1;
        
        if (minPenalty > nCurrent + yCount - yCurrent) {
            result = i + 1;
            minPenalty = nCurrent + yCount - yCurrent;
        }
    };

    return result;
};
```
```py
class Solution:
    def bestClosingTime(self, customers: str) -> int:
        
        n = len(customers)
        yCount = nCount = 0
        
        for customer in customers:
            yCount += 1 if customer == 'Y' else 0
            nCount += 1 if customer == 'N' else 0
        
        if yCount == 0:
            return 0
        if yCount == n:
            return yCount
        
        yCurrent = nCurrent = result = 0
        minPenalty = yCount

        for i in range(n):
            if customers[i] == 'Y':
                yCurrent += 1
            else:
                nCurrent += 1
            if minPenalty > nCurrent + yCount - yCurrent:
                minPenalty = nCurrent + yCount - yCurrent
                result = i + 1
        
        return result
```
