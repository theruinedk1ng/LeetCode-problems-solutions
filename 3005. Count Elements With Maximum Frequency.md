3005. Count Elements With Maximum Frequency

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxFrequencyElements = function(nums) {
    let freq = new Map();
    let maxFreq = 1;
    let result = 0;
    
    for(let num of nums){
        if(!freq.has(num))
            freq.set(num, 1);
        else{
            freq.set(num, freq.get(num) + 1);
            maxFreq = Math.max(maxFreq, freq.get(num));
        }
    }

    for(let [num, count] of freq){
        if(count === maxFreq) result += count;
    }

    return result;
};
```
