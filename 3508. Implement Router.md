/**
 * @param {number} memoryLimit
 */
var Router = function(memoryLimit) {
    this.lim = memoryLimit;
    this.head = 0;
    this.q = [];
    this.seen = new Set();
    this.dest = new Map();
};

/** 
 * @param {number} source 
 * @param {number} d 
 * @param {number} t
 * @return {boolean}
 */
Router.prototype.addPacket = function(s, d, t) {
    const key = `${s},${d},${t}`;
    if (this.seen.has(key)) return false;
    
    if (this.q.length - this.head === this.lim) {
        const [rS, rD, rT] = this.q[this.head];
        this.head++;
        this.seen.delete(`${rS},${rD},${rT}`)
        const arr = this.dest.get(rD);
        arr.shift();
        this.dest.set(rD, arr);
    }
    
    this.seen.add(key);
    this.q.push([s, d, t]);
    const arr = this.dest.get(d) || [];
    arr.push(t);
    this.dest.set(d, arr);
    
    return true;
};

/**
 * @return {number[]}
 */
Router.prototype.forwardPacket = function() {
    if (this.head + 1 > this.q.length) return [];
    
    const [s, d, t] = this.q[this.head];
    this.head++;
    this.seen.delete(`${s},${d},${t}`);
    const arr = this.dest.get(d);
    arr.shift();
    this.dest.set(d, arr);
    
    return [s, d, t];
};

/** 
 * @param {number} d 
 * @param {number} startTime 
 * @param {number} endTime
 * @return {number}
 */
Router.prototype.getCount = function(d, start, end) {
    const arr = this.dest.get(d) || [];

    if (!(arr.length) || arr[0] > end || arr.at(-1) < start) return 0;
    
    const low = findLow(start, arr);
    const high = findHigh(end, arr);
    
    return high - low;
};

function findLow(target, arr) {
    let left = 0, right = arr.length;
    
    if (arr[left] >= target) return left;
    
    while (left < right) {
        const mid = (left + right)/2 | 0;
        if (arr[mid] >= target) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }

    return left
}

function findHigh(target, arr) {
    let left = 0, right = arr.length;

    if (arr[right] <= target) return right;

    while (left < right) {
        const mid = (left + right)/2 | 0;
        if (arr[mid] <= target) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }

    return left;
}
