3354. Make Array Elements Equal to Zero

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var countValidSelections = function(nums) {
    const n = nums.length;
    let result = 0;

    function movementLeft(index){
        let direction = -1;
        const copy = [...nums];
        while(index >= 0 && index < n){
            if(copy[index] === 0)   index += direction;
            else if(copy[index] > 0){
                copy[index] -= 1;
                direction = direction * (-1);
                index += direction;
            }
        }
        return copy.every(num => num === 0);
    }

    function movementRight(index){
        let direction = 1;
        const copy = [...nums];
        while(index >= 0 && index < n){
            if(copy[index] === 0)   index += direction;
            else if(copy[index] > 0){
                copy[index] -= 1;
                direction = direction * (-1);
                index += direction;
            }
        }
        return copy.every(num => num === 0);
    }

    for(let i = 0; i < n; i ++){
        if(nums[i] === 0){
            result += movementRight(i) === true ? 1 : 0;
            result += movementLeft(i) === true ? 1 : 0;
        }
    }

    return result;
};
```
